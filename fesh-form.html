document.getElementById('feshForm').addEventListener('submit', function(e) { e.preventDefault(); const ph = parseFloat(document.querySelector('[name="ph"]').value); const hardness = parseFloat(document.querySelector('[name="hardness"]').value); const infiltration = parseFloat(document.querySelector('[name="infiltration"]').value); if (ph < 0 || ph > 14) { alert("⚠️ Soil pH must be between 0 and 14."); return; } if (hardness < 0 || hardness > 5) { alert("⚠️ Soil hardness must be between 0 and 5 MPa."); return; } if (infiltration < 0 || infiltration > 20) { alert("⚠️ Soil infiltration must be between 0 and 20 cm/hr."); return; } document.getElementById('success').style.display = 'block'; this.reset(); });
